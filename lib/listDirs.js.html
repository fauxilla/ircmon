export default async function listDirs (path, depth = 1) {
  let dirs = []
  let nodes = await readdir(path)

  nodes = nodes.map((node) => join(path, node))
  while (nodes.length) {
    const node = nodes.shift()
    const stats = await stat(node)
    if (!stats.isDirectory()) continue
    dirs.push(node)
